#include<stdio.h>
void main()
{   float a,b;
    char c;
    scanf("%f %c %f",&a,&c,&b);

    switch(c){
        case '+' :
        printf("%f",a+b);
        break;
        case '-' :
        printf("%f",a-b);
        break;
        case '*' :
        printf("%f",a*b);
        break;
        case '/' :
        printf("%f",a/b*1.0);
        break;
        default :
        printf("wrong input");
    }
}
