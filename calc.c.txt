#include <stdio.h>

int main() {

  char op;
  double fir, sec;
  printf("Enter an operator (+, -, *, /): ");
  scanf("%c", &op);
  printf("Enter two operands: ");
  scanf("%lf %lf", &fir, &sec);

  switch (op) {
    case '+':
      printf("%.1lf + %.1lf = %.1lf", fir, sec, fir + sec);
      break;
    case '-':
      printf("%.1lf - %.1lf = %.1lf", fir, sec, fir - sec);
      break;
    case '*':
      printf("%.1lf * %.1lf = %.1lf", fir, sec, fir * sec);
      break;
    case '/':
      printf("%.1lf / %.1lf = %.1lf", fir, sec, fir / sec);
      break;

    default:
      printf("Error! operator is not correct");
  }

  return 0;
}
